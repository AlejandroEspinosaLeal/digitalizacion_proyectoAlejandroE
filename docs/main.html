Module main

Backend Main Module.

This module sets up the FastAPI application, database connections, Authentication using JWT and Bcrypt, and an asynchronous WebSocket manager to handle real-time connection from agent endpoints.

@app.websocket("/ws/events/{device_id}")
async def agent_socket(websocket: WebSocket, device_id: str, db: Session = Depends(get_db)):
    await manager.connect(device_id, websocket)
    try:
        while True:
            # Receive Real-Time JSON movement events from the local agent
            data = await websocket.receive_text()
            event_json = json.loads(data)
            
            # Persist event into the SQLite history log
            new_event = FileEvent(
                device_id=device_id,
                filename=event_json.get("filename"),
                source_path=event_json.get("source_path"),
                dest_path=event_json.get("dest_path"),
                timestamp=datetime.utcnow()
            )
            
            with Session(engine) as session:
                session.add(new_event)
                session.commit()

    except WebSocketDisconnect:
        manager.disconnect(device_id)
    except Exception as e:
        manager.disconnect(device_id)

@app.post("/auth/token")
def login(user_credentials: User, background_tasks: BackgroundTasks, db: Session = Depends(get_db)):
    user = db.exec(select(User).where(User.email == user_credentials.email)).first()
    if not user:
        raise HTTPException(status_code=401, detail="Invalid credentials")
    
    if user.failed_login_attempts >= 3:
        code = str(random.randint(100000, 999999))
        user.verification_code = code
        db.commit()
        background_tasks.add_task(send_verification_email, user.email, code, True)
        return {"status": "requires_reset", "message": "You have exceeded login attempts. Check your email to recover your account."}
        
    if not verify_password(user_credentials.hashed_password, user.hashed_password):
        user.failed_login_attempts += 1
        db.commit()
        raise HTTPException(status_code=401, detail="Invalid credentials")
    
    code = str(random.randint(100000, 999999))
    user.verification_code = code
    user.failed_login_attempts = 0
    db.commit()
    background_tasks.add_task(send_verification_email, user.email, code, False)
    
    return {"status": "pending_verification", "message": "Verification code sent to email"}

@app.post("/devices/log/report")
def receive_log_report(req_data: LogReportRequest, request: Request, background_tasks: BackgroundTasks, db: Session = Depends(get_db)):
    auth_header = request.headers.get("Authorization")
    if not auth_header or not auth_header.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="No autorizado")
    
    token = auth_header.split(" ")[1]
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        email = payload.get("sub")
    except JWTError:
        raise HTTPException(status_code=401, detail="Token inválido")
        
    user = db.exec(select(User).where(User.email == email)).first()
    if not user:
        raise HTTPException(status_code=401, detail="Usuario no encontrado")

    background_tasks.add_task(send_report_email_task, user.email, req_data.device_id, req_data.folder, req_data.logs)
        
    return {"status": "success", "message": f"Report sent to {user.email}"}

@app.post("/auth/register", status_code=status.HTTP_201_CREATED)
def register(user_data: User, db: Session = Depends(get_db)):
    """Registers a new User generating Bcrypt hashes."""
    # Validate unique email
    existing = db.exec(select(User).where(User.email == user_data.email)).first()
    if existing:
        raise HTTPException(status_code=400, detail="Email is already registered")
    
    user_data.hashed_password = get_password_hash(user_data.hashed_password)
    db.add(user_data)
    db.commit()
    db.refresh(user_data)
    return {"message": "User successfully created", "id": user_data.id}

@app.post("/auth/reset-password")
def reset_password(req: ResetPasswordRequest, db: Session = Depends(get_db)):
    user = db.exec(select(User).where(User.email == req.email)).first()
    if not user or user.verification_code != req.code:
        raise HTTPException(status_code=401, detail="Invalid or expired code.")
    
    user.hashed_password = get_password_hash(req.new_password)
    user.verification_code = None
    user.failed_login_attempts = 0
    db.commit()
    
    access_token = create_access_token(subject=user.email)
    return {"access_token": access_token, "token_type": "bearer"}

def send_report_email_task(to_email: str, device_id: str, folder: str, logs: List[Dict[str, Any]]):
    if not settings.SMTP_USER or not settings.SMTP_PASSWORD:
        return
        
    now = datetime.now()
    date_str = now.strftime("%d-%m-%Y")
    subject = f"log dia {date_str}"
    
    template = templates.get_template("email_report.html")
    html_content = template.render({"device_id": device_id, "folder": folder, "logs": logs})
    
    try:
        msg = EmailMessage()
        msg['Subject'] = subject
        msg['From'] = settings.SMTP_USER
        msg['To'] = to_email
        msg.set_content("Por favor, activa la vista HTML para ver este reporte de organización.")
        msg.add_alternative(html_content, subtype='html')

        server = smtplib.SMTP('smtp.gmail.com', 587)
        server.starttls()
        server.login(settings.SMTP_USER, settings.SMTP_PASSWORD)
        server.send_message(msg)
        server.quit()
        logger.info(f"Reporte diario enviado a {to_email}")
    except Exception as e:
        logger.error(f"Error enviando reporte diario a {to_email}: {e}")

def send_verification_email(to_email: str, code: str, is_recovery: bool = False):
    if not settings.SMTP_USER or not settings.SMTP_PASSWORD:
        print(f"⚠️ SMTP not configured in .env. The code for {to_email} is: {code}")
        return
        
    now = datetime.now()
    date_str = now.strftime("%d-%m-%Y")
    
    if is_recovery:
        subject = f"restablecer contrasena dia {date_str}"
        title = "Recuperación de Contraseña"
        message_body = "Has solicitado restablecer tu contraseña. Utiliza el siguiente código de 6 dígitos para continuar. Si no fuiste tú, puedes ignorar este mensaje de seguridad."
    else:
        subject = "Código de Verificación - File Sorter"
        title = "Iniciando Sesión"
        message_body = "Se ha detectado un intento de inicio de sesión. Utiliza el siguiente código de 6 dígitos para acceder de forma segura a tu cuenta."

    template = templates.get_template("email_verification.html")
    html_content = template.render({"title": title, "message_body": message_body, "code": code})

    try:
        msg = EmailMessage()
        msg['Subject'] = subject
        msg['From'] = settings.SMTP_USER
        msg['To'] = to_email
        msg.set_content(f"Tu código es: {code}. Por favor activa la vista HTML para ver este mensaje.")
        msg.add_alternative(html_content, subtype='html')

        server = smtplib.SMTP('smtp.gmail.com', 587)
        server.starttls()
        server.login(settings.SMTP_USER, settings.SMTP_PASSWORD)
        server.send_message(msg)
        server.quit()
        logger.info(f"Correo HTML de verificación enviado con éxito a {to_email}")
    except Exception as e:
        logger.error(f"Error enviando correo SMTP a {to_email}: {e}")

@app.get("/devices/sync/{device_id}", response_model=List[SortingRule])
def sync_rules(device_id: str, db: Session = Depends(get_db)):
    """The local Python agent polls this endpoint to fetch Organization Rules."""
    # First, update the device's online status
    device = db.get(Device, device_id)
    if not device:
        # Register a new device if unseen
        device = Device(id=device_id, name=f"PC-{device_id[:4]}", user_id=1) 
        db.add(device)
    
    device.is_online = True
    device.last_sync = datetime.utcnow()
    db.commit()
    
    # Return all active sorting rules assigned to this device
    rules = db.exec(select(SortingRule).where(SortingRule.device_id == device_id)).all()
    return rules
